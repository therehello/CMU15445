78template <class T1, class T2>
79auto IntegerParentType::AddValue(const Value &left, const Value &right) const -> Value {
80 auto x = left.GetAs<T1>();
81 auto y = right.GetAs<T2>();
82 auto sum1 = static_cast<T1>(x + y);
83 auto sum2 = static_cast<T2>(x + y);
84
85 if ((x + y) != sum1 && (x + y) != sum2) {
86 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
87 }
88 // Overflow detection
89 if (sizeof(x) >= sizeof(y)) {
90 if ((x > 0 && y > 0 && sum1 < 0) || (x < 0 && y < 0 && sum1 > 0)) {
91 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
92 }
93 return Value(left.GetTypeId(), sum1);
94 }
95 if ((x > 0 && y > 0 && sum2 < 0) || (x < 0 && y < 0 && sum2 > 0)) {
96 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
97 }
98 return Value(right.GetTypeId(), sum2);
99}

123template <class T1, class T2>
124auto IntegerParentType::MultiplyValue(const Value &left, const Value &right) const -> Value {
125 auto x = left.GetAs<T1>();
126 auto y = right.GetAs<T2>();
127 auto prod1 = static_cast<T1>(x * y);
128 auto prod2 = static_cast<T2>(x * y);
129 if ((x * y) != prod1 && (x * y) != prod2) {
130 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
131 }
132 // Overflow detection
133 if (sizeof(x) >= sizeof(y)) {
134 if ((y != 0 && prod1 / y != x)) {
135 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
136 }
137 return Value(left.GetTypeId(), prod1);
138 }
139 if (y != 0 && prod2 / y != x) {
140 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
141 }
142 return Value(right.GetTypeId(), prod2);
143}

147 auto x = left.GetAs<T1>();
148 auto y = right.GetAs<T2>();

151 }

153 auto quot2 = static_cast<T2>(x / y);

156 }
