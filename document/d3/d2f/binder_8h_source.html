218 class ContextGuard {
219 public:
220 explicit ContextGuard(const BoundTableRef **scope, const CTEList **cte_scope) {
221 old_scope_ = *scope;
222 scope_ptr_ = scope;
223 old_cte_scope_ = *cte_scope;
224 cte_scope_ptr_ = cte_scope;
225 *scope = nullptr;
226 // do not reset CTE scope because we want to use those from parents
227 }
228 ~ContextGuard() {
229 *scope_ptr_ = old_scope_;
230 *cte_scope_ptr_ = old_cte_scope_;
231 }
232
233 DISALLOW_COPY_AND_MOVE(ContextGuard);
234
235 private:
236 const BoundTableRef *old_scope_;
237 const BoundTableRef **scope_ptr_;
238 const CTEList *old_cte_scope_;
239 const CTEList **cte_scope_ptr_;
240 };
