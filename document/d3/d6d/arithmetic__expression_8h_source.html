32enum class ArithmeticType { Plus, Minus };
33
37class ArithmeticExpression : public AbstractExpression {
38 public:
40 ArithmeticExpression(AbstractExpressionRef left, AbstractExpressionRef right, ArithmeticType compute_type)
41 : AbstractExpression({std::move(left), std::move(right)}, TypeId::INTEGER), compute_type_{compute_type} {
42 if (GetChildAt(0)->GetReturnType() != TypeId::INTEGER || GetChildAt(1)->GetReturnType() != TypeId::INTEGER) {
43 throw bustub::NotImplementedException("only support integer for now");
44 }
45 }
46
47 auto Evaluate(const Tuple *tuple, const Schema &schema) const -> Value override {
48 Value lhs = GetChildAt(0)->Evaluate(tuple, schema);
49 Value rhs = GetChildAt(1)->Evaluate(tuple, schema);
50 auto res = PerformComputation(lhs, rhs);
51 if (res == std::nullopt) {
52 return ValueFactory::GetNullValueByType(TypeId::INTEGER);
53 }
54 return ValueFactory::GetIntegerValue(*res);
55 }
56
57 auto EvaluateJoin(const Tuple *left_tuple, const Schema &left_schema, const Tuple *right_tuple,
58 const Schema &right_schema) const -> Value override {
59 Value lhs = GetChildAt(0)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
60 Value rhs = GetChildAt(1)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
61 auto res = PerformComputation(lhs, rhs);
62 if (res == std::nullopt) {
63 return ValueFactory::GetNullValueByType(TypeId::INTEGER);
64 }
65 return ValueFactory::GetIntegerValue(*res);
66 }
67
69 auto ToString() const -> std::string override {
70 return fmt::format("({}{}{})", *GetChildAt(0), compute_type_, *GetChildAt(1));
71 }
72
73 BUSTUB_EXPR_CLONE_WITH_CHILDREN(ArithmeticExpression);
74
75 ArithmeticType compute_type_;
76
77 private:
78 auto PerformComputation(const Value &lhs, const Value &rhs) const -> std::optional<int32_t> {
79 if (lhs.IsNull() || rhs.IsNull()) {
80 return std::nullopt;
81 }
82 switch (compute_type_) {
83 case ArithmeticType::Plus:
84 return lhs.GetAs<int32_t>() + rhs.GetAs<int32_t>();
85 case ArithmeticType::Minus:
86 return lhs.GetAs<int32_t>() - rhs.GetAs<int32_t>();
87 default:
88 UNREACHABLE("Unsupported arithmetic type.");
89 }
90 }
91};
92} // namespace bustub
