89class SimpleStreamWriter : public ResultWriter {
90 public:
91 explicit SimpleStreamWriter(std::ostream &stream, bool disable_header = false, const char *separator = "\t")
92 : disable_header_(disable_header), stream_(stream), separator_(separator) {}
93 static auto BoldOn(std::ostream &os) -> std::ostream & { return os << "\e[1m"; }
94 static auto BoldOff(std::ostream &os) -> std::ostream & { return os << "\e[0m"; }
95 void WriteCell(const std::string &cell) override { stream_ << cell << separator_; }
96 void WriteHeaderCell(const std::string &cell) override {
97 if (!disable_header_) {
98 stream_ << BoldOn << cell << BoldOff << separator_;
99 }
100 }
101 void BeginHeader() override {}
102 void EndHeader() override {
103 if (!disable_header_) {
104 stream_ << std::endl;
105 }
106 }
107 void BeginRow() override {}
108 void EndRow() override { stream_ << std::endl; }
109 void BeginTable(bool simplified_output) override {}
110 void EndTable() override {}
111
112 bool disable_header_;
113 std::ostream &stream_;
114 std::string separator_;
115};

131class HtmlWriter : public ResultWriter {
132 auto Escape(const std::string &data) -> std::string {
133 std::string buffer;
134 buffer.reserve(data.size());
135 for (const char &ch : data) {
136 switch (ch) {
137 case '&':
138 buffer.append("&amp;");
139 break;
140 case '\"':
141 buffer.append("&quot;");
142 break;
143 case '\'':
144 buffer.append("&apos;");
145 break;
146 case '<':
147 buffer.append("&lt;");
148 break;
149 case '>':
150 buffer.append("&gt;");
151 break;
152 default:
153 buffer.push_back(ch);
154 break;
155 }
156 }
157 return buffer;
158 }
159
160 public:
161 void WriteCell(const std::string &cell) override {
162 std::cout << cell;
163 if (!simplified_output_) {
164 ss_ << "<td>" << Escape(cell) << "</td>";
165 } else {
166 ss_ << Escape(cell);
167 }
168 }
169 void WriteHeaderCell(const std::string &cell) override {
170 if (!simplified_output_) {
171 ss_ << "<td>" << Escape(cell) << "</td>";
172 } else {
173 ss_ << Escape(cell);
174 }
175 }
176 void BeginHeader() override {
177 if (!simplified_output_) {
178 ss_ << "<thead><tr>";
179 }
180 }
181 void EndHeader() override {
182 if (!simplified_output_) {
183 ss_ << "</tr></thead>";
184 }
185 }
186 void BeginRow() override {
187 if (!simplified_output_) {
188 ss_ << "<tr>";
189 }
190 }
191 void EndRow() override {
192 if (!simplified_output_) {
193 ss_ << "</tr>";
194 }
195 }
196 void BeginTable(bool simplified_output) override {
197 simplified_output_ = simplified_output;
198 if (!simplified_output_) {
199 ss_ << "<table>";
200 } else {
201 ss_ << "<div>";
202 }
203 }
204 void EndTable() override {
205 if (!simplified_output_) {
206 ss_ << "</table>";
207 } else {
208 ss_ << "</div>";
209 }
210 }
211 std::stringstream ss_;
212};
