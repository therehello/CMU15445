57struct UndoLink {
58 /* Previous version can be found in which txn */
59 txn_id_t prev_txn_{INVALID_TXN_ID};
60 /* The log index of the previous version in `prev_txn_` */
61 int prev_log_idx_{0};
62
63 friend auto operator==(const UndoLink &a, const UndoLink &b) {
64 return a.prev_txn_ == b.prev_txn_ && a.prev_log_idx_ == b.prev_log_idx_;
65 }
66
67 friend auto operator!=(const UndoLink &a, const UndoLink &b) { return !(a == b); }
68
69 /* Checks if the undo link points to something. */
70 auto IsValid() const -> bool { return prev_txn_ != INVALID_TXN_ID; }
71};
