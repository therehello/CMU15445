33class ComparisonExpression : public AbstractExpression {
34 public:
36 ComparisonExpression(AbstractExpressionRef left, AbstractExpressionRef right, ComparisonType comp_type)

39 auto Evaluate(const Tuple *tuple, const Schema &schema) const -> Value override {
40 Value lhs = GetChildAt(0)->Evaluate(tuple, schema);
41 Value rhs = GetChildAt(1)->Evaluate(tuple, schema);
42 return ValueFactory::GetBooleanValue(PerformComparison(lhs, rhs));
43 }
44
45 auto EvaluateJoin(const Tuple *left_tuple, const Schema &left_schema, const Tuple *right_tuple,
46 const Schema &right_schema) const -> Value override {
47 Value lhs = GetChildAt(0)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
48 Value rhs = GetChildAt(1)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
49 return ValueFactory::GetBooleanValue(PerformComparison(lhs, rhs));
50 }
51
53 auto ToString() const -> std::string override {
54 return fmt::format("({}{}{})", *GetChildAt(0), comp_type_, *GetChildAt(1));
55 }
56
57 BUSTUB_EXPR_CLONE_WITH_CHILDREN(ComparisonExpression);
58
59 ComparisonType comp_type_;
60
61 private:
62 auto PerformComparison(const Value &lhs, const Value &rhs) const -> CmpBool {
63 switch (comp_type_) {
64 case ComparisonType::Equal:
65 return lhs.CompareEquals(rhs);
66 case ComparisonType::NotEqual:
67 return lhs.CompareNotEquals(rhs);
68 case ComparisonType::LessThan:
69 return lhs.CompareLessThan(rhs);
70 case ComparisonType::LessThanOrEqual:
71 return lhs.CompareLessThanEquals(rhs);
72 case ComparisonType::GreaterThan:
73 return lhs.CompareGreaterThan(rhs);
74 case ComparisonType::GreaterThanOrEqual:
75 return lhs.CompareGreaterThanEquals(rhs);
76 default:
77 BUSTUB_ASSERT(false, "Unsupported comparison type.");
78 }
79 }
80};
