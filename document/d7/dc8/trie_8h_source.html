99// A Trie is a data structure that maps strings to values of type T. All operations on a Trie should not
100// modify the trie itself. It should reuse the existing nodes as much as possible, and create new nodes to
101// represent the new trie.
102//
103// You are NOT allowed to remove any `const` in this project, or use `mutable` to bypass the const checks.
104class Trie {
105 private:
106 // The root of the trie.
107 std::shared_ptr<const TrieNode> root_{nullptr};
108
109 // Create a new trie with the given root.
110 explicit Trie(std::shared_ptr<const TrieNode> root) : root_(std::move(root)) {}
111
112 public:
113 // Create an empty trie.
114 Trie() = default;
115
116 // Get the value associated with the given key.
117 // 1. If the key is not in the trie, return nullptr.
118 // 2. If the key is in the trie but the type is mismatched, return nullptr.
119 // 3. Otherwise, return the value.
120 template <class T>
121 auto Get(std::string_view key) const -> const T *;
122
123 // Put a new key-value pair into the trie. If the key already exists, overwrite the value.
124 // Returns the new trie.
125 template <class T>
126 auto Put(std::string_view key, T value) const -> Trie;
127
128 // Remove the key from the trie. If the key does not exist, return the original trie.

132 // Get the root of the trie, should only be used in test cases.
