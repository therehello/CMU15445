41 SimpleAggregationHashTable(const std::vector<AbstractExpressionRef> &agg_exprs,
42 const std::vector<AggregationType> &agg_types)

46 auto GenerateInitialAggregateValue() -> AggregateValue {
47 std::vector<Value> values{};
48 for (const auto &agg_type : agg_types_) {
49 switch (agg_type) {
50 case AggregationType::CountStarAggregate:
51 // Count start starts at zero.
52 values.emplace_back(ValueFactory::GetIntegerValue(0));
53 break;
54 case AggregationType::CountAggregate:
55 case AggregationType::SumAggregate:
56 case AggregationType::MinAggregate:
57 case AggregationType::MaxAggregate:
58 // Others starts at null.
59 values.emplace_back(ValueFactory::GetNullValueByType(TypeId::INTEGER));
60 break;
61 }
62 }
63 return {values};
64 }
65
73 void CombineAggregateValues(AggregateValue *result, const AggregateValue &input) {
74 for (uint32_t i = 0; i < agg_exprs_.size(); i++) {
75 switch (agg_types_[i]) {
76 case AggregationType::CountStarAggregate:
77 case AggregationType::CountAggregate:
78 case AggregationType::SumAggregate:
79 case AggregationType::MinAggregate:
80 case AggregationType::MaxAggregate:
81 break;
82 }
83 }
84 }

91 void InsertCombine(const AggregateKey &agg_key, const AggregateValue &agg_val) {
92 if (ht_.count(agg_key) == 0) {
93 ht_.insert({agg_key, GenerateInitialAggregateValue()});
94 }
95 CombineAggregateValues(&ht_[agg_key], agg_val);
96 }
