48 template <typename C, typename S, typename Func>
49 static auto Join(const C &input, S count, const std::string &separator, Func f) -> std::string {
50 // The result
51 std::string result;
52
53 // If the input isn't empty, append the first element. We do this so we
54 // don't need to introduce an if into the loop.
55 if (count > 0) {
56 result += f(input[0]);
57 }
58
59 // Append the remaining input components, after the first
60 for (size_t i = 1; i < count; i++) {
61 result += separator + f(input[i]);
62 }
63
64 return result;
65 }
