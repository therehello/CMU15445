78 Exception(ExceptionType exception_type, const std::string &message, bool print = true)
79 : std::runtime_error(message), type_(exception_type) {
80#ifndef NDEBUG
81 if (print && !global_disable_execution_exception_print.load()) {
82 std::string exception_message =
83 "\nException Type :: " + ExceptionTypeToString(type_) + ", Message :: " + message + "\n\n";
84 std::cerr << exception_message;
85 }
86#endif
87 }

93 static auto ExceptionTypeToString(ExceptionType type) -> std::string {
94 switch (type) {
95 case ExceptionType::INVALID:

97 case ExceptionType::OUT_OF_RANGE:

99 case ExceptionType::CONVERSION:
100 return "Conversion";
101 case ExceptionType::UNKNOWN_TYPE:
102 return "Unknown Type";
103 case ExceptionType::DECIMAL:
104 return "Decimal";

106 return "Mismatch Type";
107 case ExceptionType::DIVIDE_BY_ZERO:
108 return "Divide by Zero";
109 case ExceptionType::INCOMPATIBLE_TYPE:
110 return "Incompatible type";
111 case ExceptionType::OUT_OF_MEMORY:
112 return "Out of Memory";
113 case ExceptionType::NOT_IMPLEMENTED:
114 return "Not implemented";
115 case ExceptionType::EXECUTION:
116 return "Execution";
117 default:
118 return "Unknown";
119 }
