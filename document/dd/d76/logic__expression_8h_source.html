37class LogicExpression : public AbstractExpression {
38 public:
40 LogicExpression(AbstractExpressionRef left, AbstractExpressionRef right, LogicType logic_type)
41 : AbstractExpression({std::move(left), std::move(right)}, TypeId::BOOLEAN), logic_type_{logic_type} {
42 if (GetChildAt(0)->GetReturnType() != TypeId::BOOLEAN || GetChildAt(1)->GetReturnType() != TypeId::BOOLEAN) {
43 throw bustub::NotImplementedException("expect boolean from either side");
44 }
45 }
46
47 auto Evaluate(const Tuple *tuple, const Schema &schema) const -> Value override {
48 Value lhs = GetChildAt(0)->Evaluate(tuple, schema);
49 Value rhs = GetChildAt(1)->Evaluate(tuple, schema);
50 return ValueFactory::GetBooleanValue(PerformComputation(lhs, rhs));
51 }
52
53 auto EvaluateJoin(const Tuple *left_tuple, const Schema &left_schema, const Tuple *right_tuple,
54 const Schema &right_schema) const -> Value override {
55 Value lhs = GetChildAt(0)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
56 Value rhs = GetChildAt(1)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
57 return ValueFactory::GetBooleanValue(PerformComputation(lhs, rhs));
58 }
59
61 auto ToString() const -> std::string override {
62 return fmt::format("({}{}{})", *GetChildAt(0), logic_type_, *GetChildAt(1));
63 }
64
65 BUSTUB_EXPR_CLONE_WITH_CHILDREN(LogicExpression);
66
67 LogicType logic_type_;
68
69 private:
70 auto GetBoolAsCmpBool(const Value &val) const -> CmpBool {
71 if (val.IsNull()) {
72 return CmpBool::CmpNull;
73 }
74 if (val.GetAs<bool>()) {
75 return CmpBool::CmpTrue;
76 }
77 return CmpBool::CmpFalse;
78 }
79
80 auto PerformComputation(const Value &lhs, const Value &rhs) const -> CmpBool {
81 auto l = GetBoolAsCmpBool(lhs);
82 auto r = GetBoolAsCmpBool(rhs);
83 switch (logic_type_) {
84 case LogicType::And:
85 if (l == CmpBool::CmpFalse || r == CmpBool::CmpFalse) {
86 return CmpBool::CmpFalse;
87 }
88 if (l == CmpBool::CmpTrue && r == CmpBool::CmpTrue) {
89 return CmpBool::CmpTrue;
90 }
91 return CmpBool::CmpNull;
92 case LogicType::Or:
93 if (l == CmpBool::CmpFalse && r == CmpBool::CmpFalse) {
94 return CmpBool::CmpFalse;
95 }
96 if (l == CmpBool::CmpTrue || r == CmpBool::CmpTrue) {
97 return CmpBool::CmpTrue;
98 }
99 return CmpBool::CmpNull;
100 default:
101 UNREACHABLE("Unsupported logic type.");
102 }
103 }
104};
