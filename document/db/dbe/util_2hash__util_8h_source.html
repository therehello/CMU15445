25using hash_t = std::size_t;
26
27class HashUtil {
28 private:
29 static const hash_t PRIME_FACTOR = 10000019;
30
31 public:
32 static inline auto HashBytes(const char *bytes, size_t length) -> hash_t {
33 // https://github.com/greenplum-db/gpos/blob/b53c1acd6285de94044ff91fbee91589543feba1/libgpos/src/utils.cpp#L126
34 hash_t hash = length;
35 for (size_t i = 0; i < length; ++i) {
36 hash = ((hash << 5) ^ (hash >> 27)) ^ bytes[i];
37 }
38 return hash;
39 }
40
41 static inline auto CombineHashes(hash_t l, hash_t r) -> hash_t {
42 hash_t both[2] = {};
43 both[0] = l;
44 both[1] = r;
45 return HashBytes(reinterpret_cast<char *>(both), sizeof(hash_t) * 2);
46 }
47
48 static inline auto SumHashes(hash_t l, hash_t r) -> hash_t {
49 return (l % PRIME_FACTOR + r % PRIME_FACTOR) % PRIME_FACTOR;
50 }
51
52 template <typename T>
53 static inline auto Hash(const T *ptr) -> hash_t {
54 return HashBytes(reinterpret_cast<const char *>(ptr), sizeof(T));
55 }
56
57 template <typename T>
58 static inline auto HashPtr(const T *ptr) -> hash_t {
59 return HashBytes(reinterpret_cast<const char *>(&ptr), sizeof(void *));
60 }
61
63 static inline auto HashValue(const Value *val) -> hash_t {
64 switch (val->GetTypeId()) {
65 case TypeId::TINYINT: {
66 auto raw = static_cast<int64_t>(val->GetAs<int8_t>());
67 return Hash<int64_t>(&raw);
68 }
69 case TypeId::SMALLINT: {
70 auto raw = static_cast<int64_t>(val->GetAs<int16_t>());
71 return Hash<int64_t>(&raw);
72 }
73 case TypeId::INTEGER: {
74 auto raw = static_cast<int64_t>(val->GetAs<int32_t>());
75 return Hash<int64_t>(&raw);
76 }
77 case TypeId::BIGINT: {
78 auto raw = static_cast<int64_t>(val->GetAs<int64_t>());
79 return Hash<int64_t>(&raw);
80 }
81 case TypeId::BOOLEAN: {
82 auto raw = val->GetAs<bool>();
83 return Hash<bool>(&raw);
84 }
85 case TypeId::DECIMAL: {
86 auto raw = val->GetAs<double>();
87 return Hash<double>(&raw);
88 }
89 case TypeId::VARCHAR: {
90 auto raw = val->GetData();
91 auto len = val->GetLength();
92 return HashBytes(raw, len);
93 }
94 case TypeId::TIMESTAMP: {
95 auto raw = val->GetAs<uint64_t>();
96 return Hash<uint64_t>(&raw);
97 }
98 default: {
99 UNIMPLEMENTED("Unsupported type.");
100 }
101 }
102 }
103};
