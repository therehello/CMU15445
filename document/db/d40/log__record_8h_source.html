70 : size_(HEADER_SIZE), txn_id_(txn_id), prev_lsn_(prev_lsn), log_record_type_(log_record_type) {}
71
72 // constructor for INSERT/DELETE type
73 LogRecord(txn_id_t txn_id, lsn_t prev_lsn, LogRecordType log_record_type, const RID &rid, const Tuple &tuple)
74 : txn_id_(txn_id), prev_lsn_(prev_lsn), log_record_type_(log_record_type) {
75 if (log_record_type == LogRecordType::INSERT) {
76 insert_rid_ = rid;
77 insert_tuple_ = tuple;
78 } else {
79 assert(log_record_type == LogRecordType::APPLYDELETE || log_record_type == LogRecordType::MARKDELETE ||
80 log_record_type == LogRecordType::ROLLBACKDELETE);
81 delete_rid_ = rid;
82 delete_tuple_ = tuple;
83 }
84 // calculate log record size
85 size_ = HEADER_SIZE + sizeof(RID) + sizeof(int32_t) + tuple.GetLength();
86 }
87
88 // constructor for UPDATE type
89 LogRecord(txn_id_t txn_id, lsn_t prev_lsn, LogRecordType log_record_type, const RID &update_rid,
90 const Tuple &old_tuple, const Tuple &new_tuple)
91 : txn_id_(txn_id),
92 prev_lsn_(prev_lsn),
93 log_record_type_(log_record_type),
94 update_rid_(update_rid),
95 old_tuple_(old_tuple),
96 new_tuple_(new_tuple) {
97 // calculate log record size
98 size_ = HEADER_SIZE + sizeof(RID) + old_tuple.GetLength() + new_tuple.GetLength() + 2 * sizeof(int32_t);
99 }
100
101 // constructor for NEWPAGE type
102 LogRecord(txn_id_t txn_id, lsn_t prev_lsn, LogRecordType log_record_type, page_id_t prev_page_id, page_id_t page_id)
103 : size_(HEADER_SIZE),
104 txn_id_(txn_id),
105 prev_lsn_(prev_lsn),
106 log_record_type_(log_record_type),
107 prev_page_id_(prev_page_id),
108 page_id_(page_id) {
109 // calculate log record size, header size + sizeof(prev_page_id) + sizeof(page_id)
110 size_ = HEADER_SIZE + sizeof(page_id_t) * 2;
111 }
