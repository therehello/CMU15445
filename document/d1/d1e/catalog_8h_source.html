134 auto CreateTable(Transaction *txn, const std::string &table_name, const Schema &schema, bool create_table_heap = true)
135 -> TableInfo * {
136 if (table_names_.count(table_name) != 0) {
137 return NULL_TABLE_INFO;
138 }
139
140 // Construct the table heap
141 std::unique_ptr<TableHeap> table = nullptr;
142
143 // When create_table_heap == false, it means that we're running binder tests (where no txn will be provided) or
144 // we are running shell without buffer pool. We don't need to create TableHeap in this case.
145 if (create_table_heap) {
146 table = std::make_unique<TableHeap>(bpm_);
147 } else {
148 // Otherwise, create an empty heap only for binder tests
149 table = TableHeap::CreateEmptyHeap(create_table_heap);
150 }
151
152 // Fetch the table OID for the new table
153 const auto table_oid = next_table_oid_.fetch_add(1);
154
155 // Construct the table information
156 auto meta = std::make_unique<TableInfo>(schema, table_name, std::move(table), table_oid);
157 auto *tmp = meta.get();
158
159 // Update the internal tracking mechanisms
160 tables_.emplace(table_oid, std::move(meta));
161 table_names_.emplace(table_name, table_oid);
162 index_names_.emplace(table_name, std::unordered_map<std::string, index_oid_t>{});
163
164 return tmp;
165 }

172 auto GetTable(const std::string &table_name) const -> TableInfo * {
173 auto table_oid = table_names_.find(table_name);
174 if (table_oid == table_names_.end()) {
175 // Table not found
176 return NULL_TABLE_INFO;
177 }
178
179 auto meta = tables_.find(table_oid->second);
180 BUSTUB_ASSERT(meta != tables_.end(), "Broken Invariant");
181
182 return (meta->second).get();
183 }

211 template <class KeyType, class ValueType, class KeyComparator>
212 auto CreateIndex(Transaction *txn, const std::string &index_name, const std::string &table_name, const Schema &schema,
213 const Schema &key_schema, const std::vector<uint32_t> &key_attrs, std::size_t keysize,
214 HashFunction<KeyType> hash_function, bool is_primary_key = false,
215 IndexType index_type = IndexType::HashTableIndex) -> IndexInfo * {
216 // Reject the creation request for nonexistent table
217 if (table_names_.find(table_name) == table_names_.end()) {
218 return NULL_INDEX_INFO;
219 }
220
221 // If the table exists, an entry for the table should already be present in index_names_
222 BUSTUB_ASSERT((index_names_.find(table_name) != index_names_.end()), "Broken Invariant");
223
224 // Determine if the requested index already exists for this table
225 auto &table_indexes = index_names_.find(table_name)->second;
226 if (table_indexes.find(index_name) != table_indexes.end()) {
227 // The requested index already exists for this table
228 return NULL_INDEX_INFO;
229 }
230
231 // Construct index metdata
232 auto meta = std::make_unique<IndexMetadata>(index_name, table_name, &schema, key_attrs, is_primary_key);
233
234 // Construct the index, take ownership of metadata
235 // TODO(Kyle): We should update the API for CreateIndex
236 // to allow specification of the index type itself, not
237 // just the key, value, and comparator types
238
239 // TODO(chi): support both hash index and btree index
240 std::unique_ptr<Index> index;
241 if (index_type == IndexType::HashTableIndex) {
242 index = std::make_unique<ExtendibleHashTableIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_,
243 hash_function);
244 } else {
245 BUSTUB_ASSERT(index_type == IndexType::BPlusTreeIndex, "Unsupported Index Type");
246 index = std::make_unique<BPlusTreeIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_);
247 }
248
249 // Populate the index with all tuples in table heap
250 auto *table_meta = GetTable(table_name);
251 for (auto iter = table_meta->table_->MakeIterator(); !iter.IsEnd(); ++iter) {
252 auto [meta, tuple] = iter.GetTuple();
253 // we have to silently ignore the error here for a lot of reasons...
254 index->InsertEntry(tuple.KeyFromTuple(schema, key_schema, key_attrs), tuple.GetRid(), txn);
255 }
256
257 // Get the next OID for the new index
258 const auto index_oid = next_index_oid_.fetch_add(1);
259
260 // Construct index information; IndexInfo takes ownership of the Index itself
261 auto index_info = std::make_unique<IndexInfo>(key_schema, index_name, std::move(index), index_oid, table_name,
262 keysize, is_primary_key);
263 auto *tmp = index_info.get();
264
265 // Update internal tracking
266 indexes_.emplace(index_oid, std::move(index_info));
267 table_indexes.emplace(index_name, index_oid);
268
269 return tmp;
270 }

278 auto GetIndex(const std::string &index_name, const std::string &table_name) -> IndexInfo * {
279 auto table = index_names_.find(table_name);
280 if (table == index_names_.end()) {
281 BUSTUB_ASSERT((table_names_.find(table_name) == table_names_.end()), "Broken Invariant");
282 return NULL_INDEX_INFO;
283 }
284
285 auto &table_indexes = table->second;
286
287 auto index_meta = table_indexes.find(index_name);
288 if (index_meta == table_indexes.end()) {
289 return NULL_INDEX_INFO;
290 }
291
292 auto index = indexes_.find(index_meta->second);
293 BUSTUB_ASSERT((index != indexes_.end()), "Broken Invariant");
294
295 return index->second.get();
296 }

335 auto GetTableIndexes(const std::string &table_name) const -> std::vector<IndexInfo *> {
336 // Ensure the table exists
337 if (table_names_.find(table_name) == table_names_.end()) {
338 return std::vector<IndexInfo *>{};
339 }
340
341 auto table_indexes = index_names_.find(table_name);
342 BUSTUB_ASSERT((table_indexes != index_names_.end()), "Broken Invariant");
343
344 std::vector<IndexInfo *> indexes{};
345 indexes.reserve(table_indexes->second.size());
346 for (const auto &index_meta : table_indexes->second) {
347 auto index = indexes_.find(index_meta.second);
348 BUSTUB_ASSERT((index != indexes_.end()), "Broken Invariant");
349 indexes.push_back(index->second.get());
350 }
351
352 return indexes;
353 }
