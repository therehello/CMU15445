54 auto Execute(const AbstractPlanNodeRef &plan, std::vector<Tuple> *result_set, Transaction *txn,
55 ExecutorContext *exec_ctx) -> bool {
56 BUSTUB_ASSERT((txn == exec_ctx->GetTransaction()), "Broken Invariant");
57
58 // Construct the executor for the abstract plan node
59 auto executor = ExecutorFactory::CreateExecutor(exec_ctx, plan);
60
61 // Initialize the executor
62 auto executor_succeeded = true;
63
64 try {
65 executor->Init();
66 PollExecutor(executor.get(), plan, result_set);
67 PerformChecks(exec_ctx);
68 } catch (const ExecutionException &ex) {
69 executor_succeeded = false;
70 if (result_set != nullptr) {
71 result_set->clear();
72 }
73 }
74
75 return executor_succeeded;
76 }
77
78 void PerformChecks(ExecutorContext *exec_ctx) {
79 for (const auto &[left_executor, right_executor] : exec_ctx->GetNLJCheckExecutorSet()) {
80 auto casted_left_executor = dynamic_cast<const InitCheckExecutor *>(left_executor);
81 auto casted_right_executor = dynamic_cast<const InitCheckExecutor *>(right_executor);
82 BUSTUB_ASSERT(casted_right_executor->GetInitCount() + 1 >= casted_left_executor->GetNextCount(),
83 "nlj check failed, are you initialising the right executor every time when there is a left tuple? "
84 "(off-by-one is okay)");
85 }
86 }

95 static void PollExecutor(AbstractExecutor *executor, const AbstractPlanNodeRef &plan,
96 std::vector<Tuple> *result_set) {
97 RID rid{};
98 Tuple tuple{};
99 while (executor->Next(&tuple, &rid)) {
100 if (result_set != nullptr) {
101 result_set->push_back(tuple);
102 }
103 }
104 }
