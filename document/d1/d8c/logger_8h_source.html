170// Output log message header in this format: [type] [file:line:function] time -
171// ex: [ERROR] [somefile.cpp:123:doSome()] 2008/07/06 10:00:00 -
172inline void OutputLogHeader(const char *file, int line, const char *func, int level) {
173 time_t t = ::time(nullptr);
174 tm *curTime = localtime(&t); // NOLINT
175 char time_str[32]; // FIXME
176 ::strftime(time_str, 32, LOG_LOG_TIME_FORMAT, curTime);
177 const char *type;
178 switch (level) {
179 case LOG_LEVEL_ERROR:
180 type = "ERROR";
181 break;
182 case LOG_LEVEL_WARN:
183 type = "WARN ";
184 break;
185 case LOG_LEVEL_INFO:
186 type = "INFO ";
187 break;
188 case LOG_LEVEL_DEBUG:
189 type = "DEBUG";
190 break;
191 case LOG_LEVEL_TRACE:
192 type = "TRACE";
193 break;
194 default:
195 type = "UNKWN";
196 }
197 // PAVLO: DO NOT CHANGE THIS
198 ::fprintf(LOG_OUTPUT_STREAM, "%s [%s:%d:%s] %s - ", time_str, file, line, func, type);
199}
