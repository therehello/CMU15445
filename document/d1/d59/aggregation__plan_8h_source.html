128 auto operator()(const bustub::AggregateKey &agg_key) const -> std::size_t {
129 size_t curr_hash = 0;
130 for (const auto &key : agg_key.group_bys_) {
131 if (!key.IsNull()) {
132 curr_hash = bustub::HashUtil::CombineHashes(curr_hash, bustub::HashUtil::HashValue(&key));
133 }
134 }
135 return curr_hash;
136 }
