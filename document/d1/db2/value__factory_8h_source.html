77 static inline auto GetNullValueByType(TypeId type_id) -> Value {
78 Value ret_value;
79 switch (type_id) {
80 case TypeId::BOOLEAN:
81 ret_value = GetBooleanValue(BUSTUB_BOOLEAN_NULL);
82 break;
83 case TypeId::TINYINT:
84 ret_value = GetTinyIntValue(BUSTUB_INT8_NULL);
85 break;
86 case TypeId::SMALLINT:
87 ret_value = GetSmallIntValue(BUSTUB_INT16_NULL);
88 break;
89 case TypeId::INTEGER:
90 ret_value = GetIntegerValue(BUSTUB_INT32_NULL);
91 break;
92 case TypeId::BIGINT:
93 ret_value = GetBigIntValue(BUSTUB_INT64_NULL);
94 break;
95 case TypeId::DECIMAL:
96 ret_value = GetDecimalValue(BUSTUB_DECIMAL_NULL);
97 break;
98 case TypeId::VARCHAR:
99 ret_value = GetVarcharValue(nullptr, false, nullptr);
100 break;
101 default: {
102 throw Exception(ExceptionType::UNKNOWN_TYPE, "Attempting to create invalid null type");
103 }
104 }
105 return ret_value;
106 }

132 static inline auto CastAsBigInt(const Value &value) -> Value {
133 if (Type::GetInstance(TypeId::BIGINT)->IsCoercableFrom(value.GetTypeId())) {
134 if (value.IsNull()) {
135 return ValueFactory::GetBigIntValue(static_cast<int64_t>(BUSTUB_INT64_NULL));
136 }
137 switch (value.GetTypeId()) {
138 case TypeId::TINYINT:
139 return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<int8_t>()));
140 case TypeId::SMALLINT:
141 return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<int16_t>()));
142 case TypeId::INTEGER:
143 return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<int32_t>()));
144 case TypeId::BIGINT:
145 return ValueFactory::GetBigIntValue(value.GetAs<int64_t>());
146 case TypeId::DECIMAL: {
147 if (value.GetAs<double>() > static_cast<double>(BUSTUB_INT64_MAX) ||
148 value.GetAs<double>() < static_cast<double>(BUSTUB_INT64_MIN)) {
149 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
150 }
151 return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<double>()));
152 }
153 case TypeId::VARCHAR: {
154 std::string str = value.ToString();
155 int64_t bigint = 0;
156 try {
157 bigint = stoll(str);
158 } catch (std::out_of_range &e) {
159 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
160 } catch (std::invalid_argument &e) {
161 throw Exception("Invalid input syntax for bigint: \'" + str + "\'");
162 }
163 if (bigint > BUSTUB_INT64_MAX || bigint < BUSTUB_INT64_MIN) {
164 throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
165 }
166 return ValueFactory::GetBigIntValue(bigint);
167 }
168 default:
169 break;
170 }
171 }
172 throw Exception(Type::GetInstance(value.GetTypeId())->ToString(value) + " is not coercable to BIGINT.");
173 }

394 static inline auto CastAsTimestamp(const Value &value) -> Value {
395 if (Type::GetInstance(TypeId::TIMESTAMP)->IsCoercableFrom(value.GetTypeId())) {
396 if (value.IsNull()) {
397 return ValueFactory::GetTimestampValue(BUSTUB_TIMESTAMP_NULL);
398 }
399 switch (value.GetTypeId()) {
400 case TypeId::TIMESTAMP:
401 return ValueFactory::GetTimestampValue(value.GetAs<uint64_t>());
402 case TypeId::VARCHAR: {
403 std::string str = value.ToString();
404 if (str.length() == 22) {
405 str = str.substr(0, 19) + ".000000" + str.substr(19, 3);
406 }
407 if (str.length() != 29) {
408 throw Exception("Timestamp format error.");
409 }
410 if (str[10] != ' ' || str[4] != '-' || str[7] != '-' || str[13] != ':' || str[16] != ':' || str[19] != '.' ||
411 (str[26] != '+' && str[26] != '-')) {
412 throw Exception("Timestamp format error.");
413 }
414 bool is_digit[29] = {true, true, true, true, false, true, true, false, true, true,
415 false, true, true, false, true, true, false, true, true, false,
416 true, true, true, true, true, true, false, true, true};
417 for (int i = 0; i < 29; i++) {
418 if (is_digit[i]) {
419 if (str[i] < '0' || str[i] > '9') {
420 throw Exception("Timestamp format error.");
421 }
422 }
423 }
424 int tz = 0;
425 uint32_t year = 0;
426 uint32_t month = 0;
427 uint32_t day = 0;
428 uint32_t hour = 0;
429 uint32_t min = 0;
430 uint32_t sec = 0;
431 uint32_t micro = 0;
432 uint64_t res = 0;
433 if (sscanf(str.c_str(), "%4u-%2u-%2u %2u:%2u:%2u.%6u%3d", &year, &month, &day, &hour, &min, &sec, &micro,
434 &tz) != 8) {
435 throw Exception("Timestamp format error.");
436 }
437 if (year > 9999 || month > 12 || day > 31 || hour > 23 || min > 59 || sec > 59 || micro > 999999 ||
438 day == 0 || month == 0) {
439 throw Exception(ExceptionType::OUT_OF_RANGE, "Timestamp value out of range.");
440 }
441 uint32_t max_day[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
442 uint32_t max_day_lunar[13] = {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
443 if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
444 if (day > max_day_lunar[month]) {
445 throw Exception(ExceptionType::OUT_OF_RANGE, "Timestamp value out of range.");
446 }
447 } else if (day > max_day[month]) {
448 throw Exception(ExceptionType::OUT_OF_RANGE, "Timestamp value out of range.");
449 }
450 uint32_t timezone = tz + 12;
451 if (tz > 26) {
452 throw Exception("Timestamp format error.");
453 }
454 res += month;
455 res *= 32;
456 res += day;
457 res *= 27;
458 res += timezone;
459 res *= 10000;
460 res += year;
461 res *= 100000;
462 res += hour * 3600 + min * 60 + sec;
463 res *= 1000000;
464 res += micro;
465 return ValueFactory::GetTimestampValue(res);
466 }
467 default:
468 break;
469 }
470 }
471 throw Exception(Type::GetInstance(value.GetTypeId())->ToString(value) + " is not coercable to TIMESTAMP.");
472 }

474 static inline auto CastAsBoolean(const Value &value) -> Value {
475 if (Type::GetInstance(TypeId::BOOLEAN)->IsCoercableFrom(value.GetTypeId())) {
476 if (value.IsNull()) {
477 return ValueFactory::GetBooleanValue(BUSTUB_BOOLEAN_NULL);

481 return ValueFactory::GetBooleanValue(value.GetAs<int8_t>());
482 case TypeId::VARCHAR: {
483 std::string str = value.ToString();
484 std::transform(str.begin(), str.end(), str.begin(), ::tolower);
485 if (str == "true" || str == "1" || str == "t") {
486 return ValueFactory::GetBooleanValue(true);
487 }
488 if (str == "false" || str == "0" || str == "f") {
489 return ValueFactory::GetBooleanValue(false);
490 }
491 throw Exception("Boolean value format error.");
492 }
493 default:
494 break;
495 }
496 }
497 throw Exception(Type::GetInstance(value.GetTypeId())->ToString(value) + " is not coercable to BOOLEAN.");
