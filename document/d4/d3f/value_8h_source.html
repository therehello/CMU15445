29// A value is an abstract class that represents a view over SQL data stored in
30// some materialized state. All values have a type and comparison functions, but
31// subclasses implement other type-specific functionality.
32class Value {

46 public:
47 explicit Value(const TypeId type) : manage_data_(false), type_id_(type) { size_.len_ = BUSTUB_VALUE_NULL; }
48 // BOOLEAN and TINYINT
49 Value(TypeId type, int8_t i);
50 // DECIMAL
51 Value(TypeId type, double d);
52 Value(TypeId type, float f);
53 // SMALLINT
54 Value(TypeId type, int16_t i);
55 // INTEGER

137 // Serialize this value into the given storage space. The inlined parameter
138 // indicates whether we are allowed to inline this value into the storage
139 // space, or whether we must store only a reference to this value. If inlined
140 // is false, we may use the provided data pool to allocate space for this
141 // value, storing a reference into the allocated pool space in the storage.
142 inline void SerializeTo(char *storage) const { Type::GetInstance(type_id_)->SerializeTo(*this, storage); }
143
144 // Deserialize a value of the given type from the given storage space.
145 inline static auto DeserializeFrom(const char *storage, const TypeId type_id) -> Value {
146 return Type::GetInstance(type_id)->DeserializeFrom(storage);
147 }

151 // Create a copy of this value

173 bool manage_data_;
174 // The data type
175 TypeId type_id_;
