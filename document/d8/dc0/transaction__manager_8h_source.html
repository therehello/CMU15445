34struct VersionUndoLink {
36 UndoLink prev_;
39 bool in_progress_{false};
40
41 friend auto operator==(const VersionUndoLink &a, const VersionUndoLink &b) {
42 return a.prev_ == b.prev_ && a.in_progress_ == b.in_progress_;
43 }
44
45 friend auto operator!=(const VersionUndoLink &a, const VersionUndoLink &b) { return !(a == b); }
46
47 inline static auto FromOptionalUndoLink(std::optional<UndoLink> undo_link) -> std::optional<VersionUndoLink> {
48 if (undo_link.has_value()) {
49 return VersionUndoLink{*undo_link};
50 }
51 return std::nullopt;
52 }
53};
