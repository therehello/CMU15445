169 void Print(std::ostream &out_buf) {
170 std::vector<PrintableBPlusTree *> que = {this};
171 while (!que.empty()) {
172 std::vector<PrintableBPlusTree *> new_que;
173
174 for (auto &t : que) {
175 int padding = (t->size_ - t->keys_.size()) / 2;
176 out_buf << std::string(padding, ' ');
177 out_buf << t->keys_;
178 out_buf << std::string(padding, ' ');
179
180 for (auto &c : t->children_) {
181 new_que.push_back(&c);
182 }
183 }
184 out_buf << "\n";
185 que = new_que;
186 }
