33enum class StringExpressionType { Lower, Upper };
34
38class StringExpression : public AbstractExpression {
39 public:
40 StringExpression(AbstractExpressionRef arg, StringExpressionType expr_type)
41 : AbstractExpression({std::move(arg)}, TypeId::VARCHAR), expr_type_{expr_type} {
42 if (GetChildAt(0)->GetReturnType() != TypeId::VARCHAR) {
43 throw bustub::NotImplementedException("expect the first arg to be varchar");
44 }
45 }
46
47 auto Compute(const std::string &val) const -> std::string {
48 auto res = val;
49 if (expr_type_ == StringExpressionType::Lower) {
50 std::transform(res.begin(), res.end(), res.begin(), [](char c) { return std::tolower(c); });
51 } else {
52 std::transform(res.begin(), res.end(), res.begin(), [](char c) { return std::toupper(c); });
53 }
54 return res;
55 }
56
57 auto Evaluate(const Tuple *tuple, const Schema &schema) const -> Value override {
58 Value val = GetChildAt(0)->Evaluate(tuple, schema);
59 auto str = val.GetAs<char *>();
60 return ValueFactory::GetVarcharValue(Compute(str));
61 }
62
63 auto EvaluateJoin(const Tuple *left_tuple, const Schema &left_schema, const Tuple *right_tuple,
64 const Schema &right_schema) const -> Value override {
65 Value val = GetChildAt(0)->EvaluateJoin(left_tuple, left_schema, right_tuple, right_schema);
66 auto str = val.GetAs<char *>();
67 return ValueFactory::GetVarcharValue(Compute(str));
68 }
69
71 auto ToString() const -> std::string override { return fmt::format("{}({})", expr_type_, *GetChildAt(0)); }
72
73 BUSTUB_EXPR_CLONE_WITH_CHILDREN(StringExpression);
74
75 StringExpressionType expr_type_;
76
77 private:
78};
79} // namespace bustub
