47 // Comparison functions
48 //
49 // NOTE:
50 // We could get away with only CompareLessThan() being purely virtual, since
51 // the remaining comparison functions can derive their logic from
52 // CompareLessThan(). For example:
53 //
54 // CompareEquals(o) = !CompareLessThan(o) && !o.CompareLessThan(this)
55 // CompareNotEquals(o) = !CompareEquals(o)
56 // CompareLessThanEquals(o) = CompareLessThan(o) || CompareEquals(o)
57 // CompareGreaterThan(o) = !CompareLessThanEquals(o)
58 // ... etc. ...
59 //
60 // We don't do this for two reasons:
61 // (1) The redundant calls to CompareLessThan() may be a performance problem,
62 // and since Value is a core component of the execution engine, we want to
63 // make it as performant as possible.
64 // (2) Keep the interface consistent by making all functions purely virtual.
65 virtual auto CompareEquals(const Value &left, const Value &right) const -> CmpBool;
