123 void ProcessLatency(page_id_t page_id) {
124 uint64_t sleep_micro_sec = 1000; // for random access, 1ms latency
125 if (latency_simulator_enabled_) {
126 std::unique_lock<std::mutex> lck(latency_processor_mutex_);
127 for (auto &recent_page_id : recent_access_) {
128 if ((recent_page_id & (~0x3)) == (page_id & (~0x3))) {
129 sleep_micro_sec = 100; // for access in the same "block", 0.1ms latency
130 break;
131 }
132 if (page_id >= recent_page_id && page_id <= recent_page_id + 3) {
133 sleep_micro_sec = 100; // for sequential access, 0.1ms latency
134 break;
135 }
136 }
137 lck.unlock();
138 std::this_thread::sleep_for(std::chrono::microseconds(sleep_micro_sec));
139 }
140 }

142 void PostProcessLatency(page_id_t page_id) {
143 if (latency_simulator_enabled_) {
144 std::scoped_lock<std::mutex> lck(latency_processor_mutex_);
145 recent_access_[access_ptr_] = page_id;
146 access_ptr_ = (access_ptr_ + 1) % recent_access_.size();
147 }
148 }
